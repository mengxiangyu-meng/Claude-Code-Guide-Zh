### 1.4 团队工作流

```typescript
// team-workflow.ts
import { Claude } from '@anthropic-ai/claude-sdk';

class TeamWorkflow {
  private claude: Claude;

  constructor(apiKey: string) {
    this.claude = new Claude({ 
      apiKey,
      tools: ['read', 'write', 'edit', 'bash']
    });
  }

  // PR 审查流程
  async reviewPR(prUrl: string) {
    const changes = await this.getPRChanges(prUrl);
    const review = await this.claude.chat(`
请审查以下 PR 变更：
${changes}

按团队规范输出审查报告。
    `);
    return review;
  }

  // 代码合并流程
  async mergePR(prUrl: string) {
    const review = await this.reviewPR(prUrl);
    if (review.hasIssues) {
      return { status: 'blocked', issues: review.issues };
    }
    // 自动合并
    return await this.merge(prUrl);
  }
}
```

## 2. CI/CD 集成

### 2.1 GitHub Actions

**.github/workflows/claude.yml**
```yaml
name: Claude Code CI

on:
  pull_request:
    branches: [main]
  push:
    branches: [main]

jobs:
  claude-ci:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      
      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '22'
          
      - name: Install Claude Code
        run: |
          curl -s https://api.anthropic.com | bash
          
      - name: Run Claude Code checks
        env:
          ANTHROPIC_API_KEY: ${{ secrets.ANTHROPIC_API_KEY }}
        run: |
          claude --print "
          1. 检查代码风格
          2. 运行测试
          3. 生成报告
          "
```

### 2.2 本地 CI 流程

```bash
#!/bin/bash
# claude-ci.sh

echo "🤖 开始 Claude Code CI..."

# 1. 代码审查
echo "📝 代码审查..."
claude @review .

# 2. 安全扫描
echo "🔒 安全扫描..."
claude @security .

# 3. 测试
echo "🧪 运行测试..."
npm test

# 4. 构建
echo "📦 构建..."
npm run build

echo "✅ CI 完成"
```

### 2.3 自动化部署

```yaml
# .github/workflows/deploy.yml
name: Deploy

on:
  push:
    branches: [main]

jobs:
  deploy:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      
      - name: Deploy with Claude
        env:
          ANTHROPIC_API_KEY: ${{ secrets.ANTHROPIC_API_KEY }}
        run: |
          claude --print "
          1. 构建项目
          2. 运行冒烟测试
          3. 部署到生产环境
          "
```

## 4. 性能优化

### 4.1 减少 Token 消耗

```bash
# 使用懒加载
claude @github --lazy

# 精确指定文件
claude "审查 src/auth/login.py"

# 排除不需要的文件
claude --no-read node_modules,dist "分析代码"
```

### 4.2 优化上下文

```typescript
// 只加载需要的上下文
agent.addContext({
  type: 'file',
  path: 'src/main.ts'  // 而不是整个 src/
});

// 定期清理
agent.clearContext();
```

### 4.3 缓存优化

```typescript
// 启用缓存
const claude = new Claude({
  apiKey,
  cache: true,
  cacheDir: './.claude-cache'
});
```

### 4.4 并行处理

```typescript
// 并行执行多个任务
const [review, test, doc] = await Promise.all([
  agent.run('@review .'),
  agent.run('@unit-test .'),
  agent.run('@readme')
]);
```

## 5. 安全最佳实践

### 5.1 API Key 安全

```bash
# 使用环境变量
export ANTHROPIC_API_KEY="sk-..."

# 不要写入配置文件
# ❌ "apiKey": "sk-..." 在配置文件中

# 使用 .env 文件
echo "ANTHROPIC_API_KEY=sk-..." > .env
```

### 5.2 敏感信息保护

```typescript
// Hook 检查敏感信息
const preToolUseHook = `
if [[ "$CLAUDE_TOOL_INPUT" == *"password"* ]]; then
  echo "⚠️ 检测到敏感信息"
  exit 1
fi
`;
```

### 5.3 权限控制

```typescript
// 限制工具使用
const claude = new Claude({
  apiKey,
  allowedTools: ['read', 'bash']  // 只允许读取和命令
});
```

### 5.4 审计日志

```typescript
// 记录操作日志
agent.on('tool:use', (tool) => {
  console.log(`[${new Date()}] ${tool.name}: ${tool.input}`);
});
```

## 6. 常见问题

### Q1: 如何提高代码生成质量？

**技巧：**
- 提供具体需求
- 给出示例代码
- 说明约束条件

```bash
# ❌ 模糊
claude "写个 API"

# ✅ 清晰
claude "用 Express.js 写一个用户注册 API，包含：
1. username, email, password 字段
2. 密码用 bcrypt 加密
3. 返回 JWT token
4. 包含 basic validation"
```

### Q2: 如何处理大型项目？

**策略：**
- 分模块处理
- 使用懒加载
- 定期清理上下文

### Q3: 如何保证代码安全？

**实践：**
- 不在代码中硬编码密钥
- 使用 Hook 检查敏感操作
- 限制危险工具权限

### Q4: 如何优化成本？

**方法：**
- 使用 Sonnet 模型而非 Opus
- 减少不必要的上下文
- 启用缓存

### Q5: 团队如何统一配置？

**方案：**
- 使用共享配置文件
- 统一的 Commands 和 Skills
- 定期同步更新

### Q6: 如何处理 API 错误？

```typescript
try {
  const result = await agent.run('任务');
} catch (error) {
  if (error.code === 'rate_limit') {
    // 等待后重试
    await sleep(60000);
    return await agent.run('任务');
  }
  throw error;
}
```
